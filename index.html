To add this chat widget to your website, copy the following code snippet and paste it before the closing </body> tag:

<script>
  window.addEventListener('DOMContentLoaded', function() {
    initChatWidget({
      webhookUrl: 'YOUR_N8N_WEBHOOK_URL',
